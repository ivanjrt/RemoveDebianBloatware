# RemoveDebianBloatware

```
#Libre
sudo apt-get remove --purge libreoffice*
sudo apt-get clean
sudo apt-get autoremove

#Transmision
sudo apt-get purge transmission-gtk

#ThunderBird
sudo apt-get purge thunderbird*

#Celluloid
sudo apt remove celluloid

#PIX
sudo apt-get remove pix

#Hypnotix
sudo apt-get remove pix

#Drawing
sudo apt-get remove pix

#Dictionary
sudo apt-get remove xfce4-dict

#HexChat
sudo apt-get remove hexchat

#SimpleScan
sudo apt-get remove simple-scan

#RhythmBox
sudo apt-get remove rhythmbox

#WebApp Manager
sudo apt-get remove webapp-manager

#Firefox
sudo apt-get remove firefox
```
